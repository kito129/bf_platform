import {Component, Input, OnInit} from '@angular/core';
import {Trade} from '../../../model/report/trade';
import {TradeBets} from '../../../model/report/tradeBets';
import {TennisPoint} from '../../../model/point/tennisPoint';

@Component({
  selector: 'app-trade-bets-view',
  templateUrl: './trade-bets-view.component.html',
})
export class TradeBetsViewComponent implements OnInit {

  @Input() trade: Trade

  bets: TradeBets[] = []

  constructor() { }

  ngOnInit(): void {

    // add back
    // tslint:disable-next-line:prefer-for-of
    for(let i=0; i< this.trade.trade.trades.back.length; i++){
      this.bets.push({
        type: 'back',
        selectionN: this.trade.trade.trades.back[i].selectionN,
        selectionName: this.trade.trade.selections[this.trade.trade.trades.back[i].selectionN].runnerName,
        odds: this.trade.trade.trades.back[i].odds,
        stake: this.trade.trade.trades.back[i].stake,
        toWin: this.trade.trade.trades.back[i].stake *(this.trade.trade.trades.back[i].odds -1),
        liability: this.trade.trade.trades.back[i].stake,
        time: this.trade.trade.trades.back[i].condition.time,
        point: this.trade.trade.trades.back[i].condition.tennis.point,
        note: this.trade.trade.trades.back[i].condition.note
      })
    }
    // add lay
    // tslint:disable-next-line:prefer-for-of
    for(let i=0; i< this.trade.trade.trades.lay.length; i++){
      this.bets.push({
        type: 'lay',
        selectionN: this.trade.trade.trades.lay[i].selectionN,
        selectionName: this.trade.trade.selections[this.trade.trade.trades.lay[i].selectionN].runnerName,
        odds: this.trade.trade.trades.lay[i].odds,
        stake: this.trade.trade.trades.lay[i].bank,
        toWin: this.trade.trade.trades.lay[i].bank,
        liability: this.trade.trade.trades.lay[i].liability,
        time: this.trade.trade.trades.lay[i].condition.time,
        point: this.trade.trade.trades.lay[i].condition.tennis.point,
        note: this.trade.trade.trades.lay[i].condition.note
      })
    }

    // sort by time
    this.bets = this.bets.sort((a,b)=>{
      return a.time - b.time
    })
  }

  getSide(index: number, side: number){

    let winBackA = 0
    let loseLayA = 0
    let winLayB = 0
    let loseBackB = 0
    for(let i =0; i<=index; i++){
      if((this.bets[i].type==='back' && this.bets[i].selectionN ===0) ){
        winBackA = winBackA + this.bets[i].toWin
        loseBackB = loseBackB - this.bets[i].stake
      }
      if((this.bets[i].type==='lay' && this.bets[i].selectionN ===0)){
        loseLayA = loseLayA - this.bets[i].liability
        winLayB = winLayB + this.bets[i].stake
      }
      if((this.bets[i].type==='lay' && this.bets[i].selectionN ===1)){
        loseLayA = loseLayA + this.bets[i].toWin
        winLayB = winLayB - this.bets[i].stake

      } if((this.bets[i].type==='back' && this.bets[i].selectionN ===1)){
        winBackA = winBackA - this.bets[i].stake
        loseBackB = loseBackB + this.bets[i].toWin
      }
    }
    if(side===0){
      if(winBackA + loseLayA>0){
        return  Math.round(((winBackA + loseLayA) * (1-0.02))*100)/100
      } else{
        return Math.round((winBackA + loseLayA)*100)/100
      }

    } else if (side===1) {
      if(winLayB + loseBackB>0) {
        return Math.round(((winLayB + loseBackB) * (1-0.02))*100)/100
      } else {
        return Math.round((winLayB + loseBackB)*100)/100
      }
    }
  }

}
