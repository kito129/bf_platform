import {Component, Input, OnInit} from '@angular/core';
import {Trade, TradeRowDetail} from "../../../../model/report/trade";

@Component({
  selector: 'app-trade-row-detail-shared',
  templateUrl: './trade-row-detail-shared.component.html',
})
export class TradeRowDetailSharedComponent implements OnInit {

  @Input() trade: Trade

  constructor() { }

  ngOnInit(): void {
  }

  containsThisRunnersBets(id: number){
    for(let back of this.trade.trade.trades.back){
      if(back.selectionN === id){
        return true
      }
    }
    for(let lay of this.trade.trade.trades.lay){
      if(lay.selectionN === id){
        return true
      }
    }
    return  false
  }


}
