import {AfterViewInit, Component, Input, OnInit} from '@angular/core';
import {createChart, LineStyle, PriceScaleMode} from 'lightweight-charts';
import { DatePipe } from '@angular/common'
import {Market} from '../../../../../model/market/market';


export interface RunnerData{
  runnerId: string,
  data: any[],
  color: string,
  name: string,
  visible: boolean
}

@Component({
  selector: 'app-market-detail-prices',
  templateUrl: './market-detail-prices.component.html',
})
export class MarketDetailPricesComponent implements OnInit, AfterViewInit {

  @Input()
  selectedMarkets: Market

  public runnersData: RunnerData[] = [];

  // initialize chart
  public TvData= [];
  private tv: any;
  private chart: any;

  constructor(public datePipe: DatePipe) {  }

  ngOnInit(): void {

    this.creatTvData();
  }

  ngAfterViewInit() {

    this.generateChart()
    this.setRunnerTvDataInChart()
  }

  // plot TV trade in div
  private generateChart(){
    this.tv = document.getElementById('TVCharts') as HTMLElement
    this.chart = createChart(this.tv, {
      width: document.getElementById('TvContainer').clientWidth-70,
      height: (document.getElementById('TvContainer').clientWidth-70)*0.43,
      localization: {
        locale: 'en-GB',
      },
      rightPriceScale: {
        scaleMargins: {
          top: 0.1,
          bottom: 0.1,
        },
        mode: PriceScaleMode.Logarithmic,
        borderColor: 'rgba(197, 203, 206, 0.4)',
        autoScale: true
      },
      layout: {
        backgroundColor: '#060c17',
        textColor: '#ffffff',
      },
      grid: {
        vertLines: {
          color: 'rgba(197, 203, 206, 0.4)',
          style: LineStyle.Dotted,
        },
        horzLines: {
          color: 'rgba(197, 203, 206, 0.4)',
          style: LineStyle.Dotted,
        },
      },
      watermark: {
        visible: true,
        fontSize: 36,
        horzAlign: 'center',
        vertAlign: 'center',
        color: 'rgba(171, 71, 188, 0.2)',
        text: this.selectedMarkets.marketInfo.name,
      },
      timeScale: {
        visible: true,
        timeVisible: true,
        secondsVisible: true
      }
    });
  }

  // generate correct TV data form runner data
  private creatTvData(){
    // generate data for all runner
    let firstColor = '#c86f6f'
    let i =0
    for (const odd of this.selectedMarkets.marketPrices.odds ){

      // increment the color for each runner
      firstColor = incrementColor(firstColor,5000)

      // initialize runner data
      const tempRunner: RunnerData = {
        runnerId: odd.runnerId,
        data: [],
        color: firstColor,
        name: this.selectedMarkets.marketSelections.runners[i].name,
        visible: true
      }

      // add odds for this runner
      for (const  o of odd.odds){
        const time: number = +o.timestamp/1000
        const temp = {
          time,
          value: o.ltp,
        }
        tempRunner.data.push(temp)
      }
      // add this runner to list of data
      this.runnersData.push(tempRunner)
      i++
    }
  }



  // set TvData to trade
  private setRunnerTvDataInChart(){
    // for runner create their TvData and append their odds
    for(const i in this.runnersData){
      // this runner series
      const runnerSeries = this.chart.addLineSeries({
        title: this.selectedMarkets.marketSelections.runners[i].name,
        /*
        autoscaleInfoProvider: () => ({
          priceRange: {
            minValue: 1,
            maxValue: 8,
          },
          margins: {
            above: 0,
            below: 0,
          },
        }),

         */
      });
      runnerSeries.setData(this.runnersData[i].data);
      runnerSeries.applyOptions({
        priceLineVisible: false,
        color: this.runnersData[i].color,
        priceLineColor: this.runnersData[i].color,
        lastValueVisible: false,
      });
      runnerSeries.createPriceLine({
        price: 1,
        color: '#919191',
        lineWidth: 2,
        lineStyle: LineStyle.Dotted,
        axisLabelVisible: false,
      });
      runnerSeries.createPriceLine({
        price: 2,
        color: '#ffbdbd',
        lineWidth: 2,
        lineStyle: LineStyle.SparseDotted,
        axisLabelVisible: false,
      });


      if(this.selectedMarkets.marketInfo.sport==='FOOTBALL'){
        runnerSeries.setMarkers([
          // inPlay
          {
            time: (this.selectedMarkets.marketUpdates.updates.inPlay + 7200000)/1000,
            position: 'inBar',
            color: 'yellow',
            shape: 'circle',
            size:2,
            id: 'inplay Runner' + i,
          },
          // start half time
          {
            time: (this.selectedMarkets.marketUpdates.updates.inPlay + 7200000 +2700000)/1000+7200,
            position: 'inBar',
            color: 'red',
            shape: 'circle',
            id: 'startHalfTime Runner' + i,
          },
          // end half time
          {
            time: (this.selectedMarkets.marketUpdates.updates.inPlay + 7200000 +2700000+900000)/1000+7200,
            position: 'inBar',
            color: 'blu',
            shape: 'circle',
            id: 'endHalfTime Runner' + i,
          },
        ]);

      } else {
        runnerSeries.setMarkers([
          // in play
          {
            time: (this.selectedMarkets.marketUpdates.updates.inPlay + 7200000 )/1000,
            position: 'inBar',
            color: 'yellow',
            size:2,
            shape: 'circle',
            id: 'inplay Runner' + i,
          },
        ]);
      }

      this.TvData.push(runnerSeries)
    }
    this.chart.timeScale().fitContent();
  }

  // support function to change runner by button
  public changeRunnerState(runnerId: string){
    let i = 0
    for (const runner of this.runnersData ){
      if( runnerId == runner.runnerId){
        if(runner.visible == true ){
          this.changeRunnerVisibility(i, false)
          break
        } else {
          this.changeRunnerVisibility(i, true)
          break
        }
      }
      i++
    }
  };

  // reset all runner to visible
  public resetClick(){
    let i = 0
    for (const runner of this.runnersData ){
      this.changeRunnerVisibility(i, true)
      i++
    }
    this.chart.timeScale().resetTimeScale();
    this.chart.timeScale().fitContent();
  }

  // visible only the winner selection price click
  public winnerOnlyClick(){
    let i = 0
    for (const runner of this.runnersData ) {
      // select only the winner
      if (this.selectedMarkets.marketSelections.runners[i].status == 'WINNER') {
        this.changeRunnerVisibility(i, true)
      } else {
        this.changeRunnerVisibility(i, false)
      }
      i++
    }
  }

  // expand chart click
  public fitClick(){
    this.chart.timeScale().resetTimeScale();
    this.chart.timeScale().fitContent();
  }


  // prematch only click
  public preMatchOnlyClick(){


    this.chart.timeScale().setVisibleRange({
      from: (this.selectedMarkets.marketUpdates.updates.open + (3600000))/1000,
      to: (this.selectedMarkets.marketUpdates.updates.inPlay  )/1000,
    })
  }

  // go to Inplay click
  public goToInPlayClick(){
    this.chart.timeScale().setVisibleRange({
      from: (this.selectedMarkets.marketUpdates.updates.inPlay )/1000,
      to: (this.selectedMarkets.marketUpdates.updates.closed + (3600000))/1000,
    })
  }

  // change the visibility for the runner in this position
  private changeRunnerVisibility(runnerPos: number, status:boolean){
    this.runnersData[runnerPos].visible = status
    this.TvData[runnerPos].applyOptions({
      visible: status,
    });

  }
}


// increment the color function
export function incrementColor(color, step){
  let colorToInt = parseInt(color.substr(1), 16),
    nStep = parseInt(step);
  if(!isNaN(colorToInt) && !isNaN(nStep)){
    colorToInt += nStep;
    let nColor = colorToInt.toString(16);
    nColor = '#' + (new Array(7-nColor.length).join('0')) + nColor;
    if(/^#[0-9a-f]{6}$/i.test(nColor)){
      return nColor;
    }
  }
  return color;
}
